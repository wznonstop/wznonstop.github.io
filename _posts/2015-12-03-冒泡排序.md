===================

    	<p>冒泡排序，时间复杂度：O(N<sup>2</sup>)</p>
    	<script>
    		var arr = [25,32,7,14,99];
    		var x = 1;
    		length = arr.length;
    		while(x <= (length - 1)){
    			for (var i = 0; i < (arr.length-x); i++) {
    				var that = arr[i];
    				var cthat = arr[i+1];
    				if(that<cthat){
    			        var val = that;
    			        arr[i] = cthat;
    			        arr[i+1] = val;
    			    }
    			};
    			x++;
    		}
    		console.log(arr);
    	</script>
